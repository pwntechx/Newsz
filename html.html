⬡ DISSIPATION ENGINE

// HTML: <div id="target">🐺</div><canvas id="particleCanvas"></canvas>

function dissipate() {
  const rect = target.getBoundingClientRect();
  const ox = (rect.left - stageRect.left) + rect.width/2;
  const oy = (rect.top - stageRect.top) + rect.height/2;
  target.style.opacity = '0';
  for(let i=0;i<40;i++) particles.push(new Particle(ox, oy));
  animateParticles();
}

class Particle {
  constructor(x,y) {
    this.x = x; this.y = y;
    this.vx = (Math.random() - 0.5) * 8;
    this.vy = (Math.random() - 1.5) * 5;
    this.life = 1;
    this.decay = 0.02;
    this.size = 2 + Math.random() * 4;
  }
  update() { this.x += this.vx; this.y += this.vy; this.vy += 0.3; this.life -= this.decay; }
  draw(ctx) { ctx.globalAlpha = this.life; ctx.fillRect(this.x, this.y, this.size, this.size); }
}
Variation — Circular particles: Replace fillRect with ctx.arc().
Variation — Color fade: Use this.color = `hsl(${Math.random() * 360}, 70%, 50%)`.

⬡ EXPANDING RINGS (528Hz BURST)
Canvas animation · expanding circle · decaying opacity · click-triggered
class Ring {
  constructor(x,y) { this.x=x; this.y=y; this.r=2; this.life=1; }
  update() { this.r+=3; this.life-=0.03; return this.life > 0 && this.r < 100; }
  draw(ctx) {
    ctx.beginPath(); ctx.arc(this.x, this.y, this.r, 0, Math.PI*2);
    ctx.strokeStyle = `rgba(212, 175, 55, ${this.life})`;
    ctx.lineWidth = 2; ctx.stroke();
  }
}

// On click: rings.push(new Ring(clickX, clickY)); animate with requestAnimationFrame.
Variation — Fill rings (fading fill): ctx.fillStyle = `rgba(212, 175, 55, ${this.life * 0.2})`; ctx.fill();
Variation — Multiple simultaneous rings: Store all in array, animate each independently.

⬡ FLOATING BUBBLES
Canvas animation · sinuous rise · sine-wave horizontal drift · random sizes
class Bubble {
  constructor(x,y) {
    this.x = x ?? canvas.width * Math.random();
    this.y = y ?? canvas.height;
    this.r = 5 + Math.random() * 15;
    this.speedY = 0.5 + Math.random() * 1;
    this.phase = Math.random() * Math.PI * 2;
  }
  update() {
    this.y -= this.speedY;
    this.phase += 0.05;
    this.x += Math.sin(this.phase) * 0.4;
    return this.y > -this.r;
  }
  draw(ctx) { ctx.arc(this.x, this.y, this.r, 0, Math.PI*2); ctx.fillStyle = `rgba(212,175,55,0.4)`; ctx.fill(); ctx.stroke(); }
}
Variation — Bubble trail: Add a smaller highlight circle inside each bubble (ctx.arc(x-3, y-3, r*0.3)).
Variation — Different colors: Use hsl(50, 70%, 60%) for golden, hsl(200, 70%, 60%) for blue.

⬡ DRIFTING SNOW
Canvas particles · random wind · downward drift · infinite loop
class Snowflake {
  constructor() {
    this.x = Math.random() * canvas.width;
    this.y = Math.random() * canvas.height;
    this.r = 1 + Math.random() * 3;
    this.speedY = 0.5 + Math.random() * 1;
    this.wind = (Math.random() - 0.5) * 0.6;
    this.opacity = 0.4 + Math.random() * 0.5;
  }
  update() {
    this.y += this.speedY; this.x += this.wind;
    if (this.y > canvas.height) { this.y = 0; this.x = Math.random() * canvas.width; }
    if (this.x < 0) this.x = canvas.width; if (this.x > canvas.width) this.x = 0;
  }
  draw(ctx) {
    ctx.beginPath(); ctx.arc(this.x, this.y, this.r, 0, Math.PI*2);
    ctx.fillStyle = `rgba(255,255,255,${this.opacity})`; ctx.fill();
  }
}
Variation — Gust of wind: Periodically multiply wind by 2-3x for a few frames.
Variation — Melting effect: Decrease radius when y is near bottom edge.
◆ Dork: javascript "snowfall" canvas requestAnimationFrame
⬡ PEACH BLOSSOMS
Canvas 5-petal flowers · rotation · sine-wave drift · spring colors
class Blossom {
  draw(ctx) {
    ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.rotation);
    ctx.beginPath();
    for (let i = 0; i < 5; i++) {
      let angle = (i / 5) * Math.PI * 2;
      let x = Math.cos(angle) * this.size;
      let y = Math.sin(angle) * this.size;
      if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
    }
    ctx.closePath(); ctx.fillStyle = this.color; ctx.fill();
    ctx.restore();
  }
}
Variation — 8-petal flowers: Change loop to for (let i = 0; i < 8; i++).
Variation — Gradient fill: Use ctx.fillStyle = "hsl(350, 80%, 70%)" for pink, hsl(50, 80%, 70%) for yellow centers.
◆ Dork: javascript "flower" canvas "rotate" "forEach" petal
